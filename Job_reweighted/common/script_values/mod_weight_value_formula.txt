mod_base_pop_job_weight_bonus = {
	base = 0
	add = modifier:pop_$JOB$_bonus_workforce_mult
	add = modifier:pop_cat_$POP_CAT$_bonus_workforce_mult
	mult = 10000
	min = -10
}

worker_job_reweight = {
	base = 1
	mult = value:job_weights_modifier|JOB|$JOB$|RESOURCE|$RESOURCE$|

	modifier = {
		add = value:mod_base_pop_job_weight_bonus|JOB|$JOB$|POP_CAT|worker|
	}
	modifier = {
		can_take_servant_job = yes
		mult = 0.25
	}

	modifier = {
		OR = {
			is_enslaved = yes
			is_shackled_robot = yes
			is_non_sapient_robot = yes
		}
		can_take_servant_job = no
		mult = 100
	}

	modifier = {
		mult = 5
		has_trait = trait_cyborg_creed_of_labor
	}

	modifier = {
		factor = 3
		exists = planet
		planet = {
			has_available_jobs = $JOB$
		}
		exists = owner
		owner = {
			has_monthly_income = {
				resource = $RESOURCE$
				value < 10
			}
		}
	}
}

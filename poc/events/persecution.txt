# POC will not include policy yet. Only will force residence.

namespace = persecution_mod

country_event = {
    id = persecution_mod.0
    hide_window = yes
	is_triggered_only = yes

    immediate = {

            limit = {
                is_pop_category = worker

                cooldown = no
            }
        }
    }
